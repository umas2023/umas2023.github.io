- 封装异步操作，避免回调地狱

{% raw %}
```js
const p = new Promise((res, rej) => {
    let state = true;
    if (state) {
    res("result");
    } else {
    rej("reject");
    }
});


p.then(
    (res) => {
    console.log("promise val:", res);
    },
    (rej) => {
    console.log("promise err:", rej);
    }
);

```
